function parseCsvLine(line: string): string[] {
  const cells: string[] = [];
  let cur = '', inQuotes = false;
  for (let i = 0; i < line.length; i++) {
    const c = line[i];
    if (inQuotes) {
      if (c === '"' && line[i + 1] === '"') { cur += '"'; i++; }
      else if (c === '"') { inQuotes = false; }
      else cur += c;
    } else if (c === '"') inQuotes = true;
    else if (c === ',') { cells.push(cur); cur = ''; }
    else cur += c;
  }
  cells.push(cur);
  return cells;
}

test.describe('CSV export formula-injection sanitization', () => {
  test('a contact name starting with = @ + - is neutralized in the exported CSV, not a live formula', async ({ page }) => {
    // Contact id 6 already has a seeded formula-injection payload (=HYPERLINK(...)) as its
    // name — fetch it live rather than hardcoding the payload string, so the test tracks
    // whatever value actually lives in the app today.

    const contacts = (await contactsResp.json()).contacts ?? (await contactsResp.json());
    const target = contacts.find((c: any) => /^[=+\-@]/.test(c.name));
    expect(target, 'expected a seeded contact with a formula-injection payload as its name').toBeTruthy();

    await page.goto('/contacts');
    const [download] = await Promise.all([
      page.waitForEvent('download'),
      page.getByRole('button', { name: /export csv/i }).click(), // TODO confirm exact accessible name against live app

    const csvPath = await download.path();
    const csv = fs.readFileSync(csvPath!, 'utf-8');

    const lines = csv.split(/\r?\n/).filter(Boolean);
    const header = parseCsvLine(lines[0]).map((h) => h.trim().toLowerCase());
    const nameCol = header.indexOf('name');
    const idCol = header.indexOf('id');
    const row = lines.slice(1).map(parseCsvLine).find((cells) => cells[idCol] === String(target.id));
    expect(row, 'expected the target contact to appear in the export').toBeTruthy();

    const exportedName = row![nameCol];

function parseCsvLine(line: string): string[] {
  const cells: string[] = [];
  let cur = '', inQuotes = false;
  for (let i = 0; i < line.length; i++) {
    const c = line[i];
    if (inQuotes) {
      if (c === '"' && line[i + 1] === '"') { cur += '"'; i++; }
      else if (c === '"') { inQuotes = false; }
      else cur += c;
    } else if (c === '"') inQuotes = true;
    else if (c === ',') { cells.push(cur); cur = ''; }
    else cur += c;
  }
  cells.push(cur);
  return cells;
}

test.describe('CSV export formula-injection sanitization', () => {

    const target = contacts.find((c: any) => /^[=+\-@]/.test(c.name));
